function runLuaHighlighter(text, keyword) { if (!text) return ''; // 修正：用正则转义处理 HTML 特殊字符 let html = text.replace(/&/g, "&") .replace(//g, ">"); // 1. 字符串高亮 html = html.replace(/(".*?")/g, '$1'); // 2. Lua 关键字 const keywords = ['local', 'if', 'then', 'end', 'and', 'or', 'else', 'elseif', 'return', 'function']; keywords.forEach(kw => { const reg = new RegExp(`\\b(${kw})\\b`, 'g'); html = html.replace(reg, '$1'); }); // 3. 宏命令高亮 const macroCommands = ['/click', '/cast', '/use', '/run', '/script', '/target', '/clearfocus']; macroCommands.forEach(cmd => { const escapedCmd = cmd.replace('/', '\\/'); const reg = new RegExp(`(${escapedCmd})\\b`, 'g'); html = html.replace(reg, '$1'); }); // 4. 自定义触发函数 const specialFuncs = ['BeeRun', 'BeelsRun', 'BeeStringFind', 'BeeUnitCastSpellName']; specialFuncs.forEach(func => { const reg = new RegExp(`\\b(${func})\\b`, 'g'); html = html.replace(reg, '$1'); }); // 5. 注释 html = html.replace(/(--.*)/g, '$1'); // 6. 搜索高亮 if (keyword) { const escapedKeyword = keyword.replace(/[-\/\\^$*+?.()|[\]{}]/g, '\\$&'); const searchReg = new RegExp(`(${escapedKeyword})`, 'gi'); html = html.replace(searchReg, '$1'); } return html; }
